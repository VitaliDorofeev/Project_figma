git status -Проверить Статус

git add . -Добавить все изменения ( за место "." название файла которорого хотим изменить)
git commit -m "Название комита" -Фиксировать изменение в git и добавить название 

git branch -Статус ветки 
git branch "название ветки" -Добваить ветку 
git branch -D "название ветки" -Удалить ветку 
git checkout "название ветки" -Перейти на ветку 
git checkout -b "название ветки" -Создать новую ветку и перейти на неё
